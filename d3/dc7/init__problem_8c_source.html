259 static void matrix_read_from_file(real_t *mtx, FILE * file, const uint32_t rows, const uint32_t cols)
260 {
261  uint32_t i; /* row number */
262  uint32_t j; /* column number */
263  uint32_t k = 0; /* matrix index (row * col) */
264  for (i = 0; i < rows; i++) {
265  for (j = 0; j < cols; j++) {
266  fscanf(file,"%lf,",&mtx[k]);
267  k++;
268  }
269  }
