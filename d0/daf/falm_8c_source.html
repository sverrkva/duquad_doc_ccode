330 static real_t dual_obj(struct Struct_FALM *s)
331 {
332  /*return: 0.5z'Hz + c'z + lambda'*(A*z - b) + 0.5*rho*norm(A*z-b) */
333 
334  // f_value = 0.5*z'Hz + c'z;
335  real_t f_value = obj(s->z_ds, s->prob->H, s->prob->c, s->temp1_dim_N);
336 
337  // f_value += lambda'*(A*z-b)
338 
339  vector_sub(s->A_z_ds,s->prob->b,s->temp3_dim_M,M); // (A*z) - b
340  f_value += vector_mul(s->lambda,s->temp3_dim_M,M);
341 
342  real_t nor = vector_norm_2(s->temp3_dim_M,M);
343  f_value += 0.5 * s->opt->rho * (nor*nor);
344 
345  return f_value;
346 }
